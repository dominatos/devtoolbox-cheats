# FOR over files / –ü–µ—Ä–µ–±–æ—Ä —Ñ–∞–π–ª–æ–≤
for f in *.log; do : > "$f"; done        # Truncate all .log files / –û—á–∏—Å—Ç–∏—Ç—å –≤—Å–µ .log —Ñ–∞–π–ª—ã
for f in *.{yml,yaml}; do echo "$f"; done # Multiple globs / –ù–µ—Å–∫–æ–ª—å–∫–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–π

# FOR over numbers / –ü–µ—Ä–µ–±–æ—Ä —á–∏—Å–µ–ª
for i in {1..5}; do echo $i; done        # 1..5 (brace expansion) / –î–∏–∞–ø–∞–∑–æ–Ω –≤ —Ñ–∏–≥—É—Ä–Ω—ã—Ö —Å–∫–æ–±–∫–∞—Ö
for ((i=1;i<=5;i++)); do echo $i; done   # C-style loop / –¶–∏–∫–ª –≤ —Å—Ç–∏–ª–µ C

# WHILE / –ü–æ–∫–∞ —É—Å–ª–æ–≤–∏–µ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è
i=1; while [ $i -le 5 ]; do echo $i; ((i++)); done  # Counter loop / –°—á—ë—Ç—á–∏–∫ –¥–æ 5
while read -r line; do echo "$line"; done < file    # Read file lines / –ß—Ç–µ–Ω–∏–µ —Ñ–∞–π–ª–∞ –ø–æ—Å—Ç—Ä–æ—á–Ω–æ

# UNTIL / –ü–æ–∫–∞ —É—Å–ª–æ–≤–∏–µ –ù–ï –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è
i=1; until [ $i -gt 5 ]; do echo $i; ((i++)); done  # Run until condition true / –í—ã–ø–æ–ª–Ω—è—Ç—å –ø–æ–∫–∞ —É—Å–ª–æ–≤–∏–µ –õ–û–ñ–ù–û

# Infinite with sleep / –ë–µ—Å–∫–æ–Ω–µ—á–Ω—ã–π —Ü–∏–∫–ª —Å –ø–∞—É–∑–æ–π
while :; do date; sleep 5; done          # Tick every 5s / –ü–µ—á–∞—Ç—å –¥–∞—Ç—ã –∫–∞–∂–¥—ã–µ 5 —Å–µ–∫—É–Ω–¥

# Safe pipeline read (no subshell var loss) / –ë–µ–∑ –ø–æ—Ç–µ—Ä–∏ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö –≤ –ø–∞–π–ø–ª–∞–π–Ω–µ
while IFS= read -r f; do echo "$f"; done < <(ls -1) # Process substitution / –ü–æ–¥—Å—Ç–∞–Ω–æ–≤–∫–∞ —á–µ—Ä–µ–∑ < <(...)
